Class: OpenFileView

OpenFileView

打开文件组件类。用于打开本地数据文件并加载到地图，目前支持打开.csv|.xls|.xlsx|.geojson|.json 格式，只支持 WGS84 经纬度坐标。

new OpenFileView(options)

Usage
// 浏览器
<script type="text/javascript" src="https://iclient.supermap.io/dist/leaflet/iclient-leaflet.js"></script>
<script>
  new L.supermap.Components.OpenFileView(options);
  
  // 弃用的写法
  L.supermap.components.openFile(options);
  
</script>

// ES6 Import
import { OpenFileView } from '@supermapgis/iclient-leaflet';
